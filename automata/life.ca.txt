AUTOMATA Life
-- Conways Life
-- Der bekannteste Zelluläre Automat
-- Geschichtliches: Conway soll "The game of life" mit Tellern auf den
-- Fliesen der Eingangshalle gespielt haben.

ARRAY 50,50
STATES
	dead : white
	alife: blue
END
NEIGHBORS
	Moore 1
END
PARAMETERS
	--Cyclic
	Plane 0
	Open
	Square 5,5,10,1
	GridColor Yellow
	-- Load "life.save"
	-- Init_Rnd
END
RULES
	1 IF CELL=1 AND (SUM(1) = 2 OR SUM(1) = 3);
	1 IF CELL=0 AND SUM(1) = 3;
	0 OTHERWISE;
END
